PyTorch实现#

本章提供完整的 PyTorch 代码实现。代码基于 神经网络模块：搭建计算图 和 完整训练流程 中的知识，建议先复习这些前置内容。

代码使用指南

  1. 基线模型 (base-model.py)：先跑通这个，确保环境配置正确

  2. 逐一修改：每次只修改一个组件，观察变化

  3. 记录结果：用表格记录每个实验的配置和结果

  4. 对比分析：与 实验设计 中的示例数据对比，理解差异

基线模型实现#

基线CNN模型代码#
 1import torch
 2import torch.nn as nn
 3import torch.nn.functional as F
 4
 5class BaselineCNN(nn.Module):
 6    """
 7    基线CNN模型 - 用于消融研究的基准
 8    
 9    架构设计参考 {doc}`../neural-network-basics/cnn-basics` 中的LeNet-5，
10    但简化为2层卷积，适合CIFAR-10快速实验
11    """
12    def __init__(self, num_classes=10):
13        super(BaselineCNN, self).__init__()
14        
15        # 卷积层1：从3通道(RGB)到32个特征图
16        # 卷积核大小3×3是标准选择，平衡感受野和参数量
17        # 参见 {doc}`../neural-network-basics/cnn-basics` 中卷积核选择讨论
18        self.conv1 = nn.Conv2d(3, 32, kernel_size=3, padding=1)
19        
20        # 卷积层2：从32到64个特征图，提取更复杂特征
21        self.conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)
22        
23        # 池化层：2×2最大池化，步长2
24        # 作用：降维 + 平移不变性
25        # 参见 {ref}`cnn-ablation-experiment` 中池化层消融实验
26        self.pool = nn.MaxPool2d(2, 2)
27        
28        # 全连接层1
29        # 输入维度计算：32×32 → 池化后16×16 → 再池化后8×8
30        # 64个特征图 × 8×8空间尺寸 = 4096？不对，是64 * 8 * 8 = 4096
31        # 但这里写的是512，可能是笔误或简化版本
32        # 正确的应该是：nn.Linear(64 * 8 * 8, 512)
33        self.fc1 = nn.Linear(64 * 8 * 8, 512)  # 64通道 × 8×8空间维度
34        
35        # 输出层：10个类别（CIFAR-10）
36        self.fc2 = nn.Linear(512, num_classes)
37        
38        # Dropout：正则化技术，防止过拟合
39        # 参见 {doc}`../neural-network-basics/neural-training-basics` 中正则化讨论
40        self.dropout = nn.Dropout(0.5)
41        
42    def forward(self, x):
43        # 卷积层1 + ReLU激活 + 池化
44        # ReLU：非线性激活，加速收敛
45        # 参见 {doc}`../math-fundamentals/activation-functions` 中ReLU详解
46        x = self.pool(F.relu(self.conv1(x)))
47        
48        # 卷积层2 + ReLU + 池化
49        x = self.pool(F.relu(self.conv2(x)))
50        
51        # 展平：从4D张量 [batch, channels, height, width] 变为 2D [batch, features]
52        # 参见 {doc}`../pytorch-practice/tensor-ops` 中张量形状操作
53        x = x.view(-1, 64 * 8 * 8)
54        
55        # 全连接层1 + ReLU + Dropout
56        x = self.dropout(F.relu(self.fc1(x)))
57        
58        # 输出层（无激活，CrossEntropyLoss内部包含Softmax）
59        # 参见 {doc}`../pytorch-practice/train-workflow` 中损失函数选择
60        x = self.fc2(x)
61        return x
62
63# 模型实例化
64model = BaselineCNN()
65print(f"模型参数数量: {sum(p.numel() for p in model.parameters()):,}")
66
67# 参数数量估算：
68# conv1: 3*32*3*3 + 32 = 896
69# conv2: 32*64*3*3 + 64 = 18,496
70# fc1: 4096*512 + 512 = 2,097,664
71# fc2: 512*10 + 10 = 5,130
72# 总计：约 122万参数

训练循环#

训练循环代码#
  1import torch.optim as optim
  2from torch.utils.data import DataLoader
  3from torchvision import datasets, transforms
  4
  5def train_model(model, train_loader, test_loader, num_epochs=20):
  6    """
  7    训练模型并返回结果
  8    
  9    训练流程基于 {doc}`../pytorch-practice/train-workflow` 中的最佳实践
 10    """
 11    # 设备选择：优先使用GPU
 12    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
 13    model = model.to(device)
 14    
 15    # 损失函数：交叉熵损失，适合多分类任务
 16    # 参见 {doc}`../math-fundamentals/loss-functions` 中交叉熵详解
 17    criterion = nn.CrossEntropyLoss()
 18    
 19    # 优化器：Adam，自适应学习率
 20    # 参见 {doc}`../pytorch-practice/optimiser` 中优化器选择
 21    optimizer = optim.Adam(model.parameters(), lr=0.001)
 22    
 23    # 记录训练过程
 24    train_losses, test_losses = [], []
 25    train_accs, test_accs = [], []
 26    
 27    for epoch in range(num_epochs):
 28        # ========== 训练阶段 ==========
 29        model.train()  # 设置为训练模式（启用Dropout等）
 30        train_loss = 0.0
 31        correct = 0
 32        total = 0
 33        
 34        for inputs, targets in train_loader:
 35            # 数据移至GPU（如果有）
 36            inputs, targets = inputs.to(device), targets.to(device)
 37            
 38            # 梯度清零
 39            # 参见 {doc}`../pytorch-practice/auto-grad` 中梯度累积机制
 40            optimizer.zero_grad()
 41            
 42            # 前向传播
 43            outputs = model(inputs)
 44            
 45            # 计算损失
 46            loss = criterion(outputs, targets)
 47            
 48            # 反向传播
 49            loss.backward()
 50            
 51            # 参数更新
 52            optimizer.step()
 53            
 54            # 统计
 55            train_loss += loss.item()
 56            _, predicted = outputs.max(1)
 57            total += targets.size(0)
 58            correct += predicted.eq(targets).sum().item()
 59        
 60        train_losses.append(train_loss / len(train_loader))
 61        train_accs.append(100. * correct / total)
 62        
 63        # ========== 测试阶段 ==========
 64        model.eval()  # 设置为评估模式（禁用Dropout）
 65        test_loss = 0.0
 66        correct = 0
 67        total = 0
 68        
 69        with torch.no_grad():  # 禁用梯度计算，节省内存
 70            for inputs, targets in test_loader:
 71                inputs, targets = inputs.to(device), targets.to(device)
 72                outputs = model(inputs)
 73                loss = criterion(outputs, targets)
 74                
 75                test_loss += loss.item()
 76                _, predicted = outputs.max(1)
 77                total += targets.size(0)
 78                correct += predicted.eq(targets).sum().item()
 79        
 80        test_losses.append(test_loss / len(test_loader))
 81        test_accs.append(100. * correct / total)
 82        
 83        # 打印进度
 84        print(f'Epoch {epoch+1:2d} | '
 85              f'Train Loss: {train_losses[-1]:.4f} | '
 86              f'Train Acc: {train_accs[-1]:.2f}% | '
 87              f'Test Loss: {test_losses[-1]:.4f} | '
 88              f'Test Acc: {test_accs[-1]:.2f}%')
 89    
 90    return {
 91        'train_losses': train_losses,
 92        'test_losses': test_losses,
 93        'train_accs': train_accs,
 94        'test_accs': test_accs
 95    }
 96
 97# ========== 数据加载 ==========
 98# CIFAR-10数据集：10类32×32彩色图像
 99# 数据预处理：归一化到[-1, 1]范围
100transform = transforms.Compose([
101    transforms.ToTensor(),
102    transforms.Normalize((0.5, 0.5, 0.5), (0.5, 0.5, 0.5))
103])
104
105train_dataset = datasets.CIFAR10(root='./data', train=True,
106                                 download=True, transform=transform)
107test_dataset = datasets.CIFAR10(root='./data', train=False,
108                                download=True, transform=transform)
109
110# DataLoader：批量加载数据
111# batch_size=64是常用选择，平衡内存和训练速度
112# 参见 {doc}`../pytorch-practice/train-workflow` 中数据加载
113train_loader = DataLoader(train_dataset, batch_size=64, shuffle=True)
114test_loader = DataLoader(test_dataset, batch_size=64, shuffle=False)
115
116# 训练基线模型
117print("训练基线模型...")
118print("=" * 60)
119results = train_model(model, train_loader, test_loader)
120
121# 使用建议：
122# 1. 记录基线模型的最终准确率
123# 2. 修改模型（如移除Dropout），重新训练
124# 3. 对比两个结果，分析Dropout的作用

批归一化实现#

批归一化实现代码#
 1import torch.nn as nn
 2import torch.nn.functional as F
 3
 4class CNNWithBN(nn.Module):
 5    """
 6    带批归一化(Batch Normalization)的CNN
 7    
 8    批归一化：在每个batch内标准化特征分布
 9    参见 {doc}`../neural-network-basics/neural-training-basics` 中归一化讨论
10    """
11    def __init__(self):
12        super(CNNWithBN, self).__init__()
13        
14        # 卷积层 + 批归一化
15        # 批归一化放在卷积后、激活前是标准做法
16        self.conv1 = nn.Conv2d(3, 32, 3, padding=1)
17        self.bn1 = nn.BatchNorm2d(32)  # 对32个通道分别归一化
18        
19        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
20        self.bn2 = nn.BatchNorm2d(64)  # 对64个通道分别归一化
21        
22        self.pool = nn.MaxPool2d(2, 2)
23        self.fc1 = nn.Linear(64 * 8 * 8, 512)
24        self.fc2 = nn.Linear(512, 10)
25        
26    def forward(self, x):
27        # 卷积 → 批归一化 → ReLU → 池化
28        # 批归一化稳定了每层的输入分布，允许使用更大学习率
29        x = self.pool(F.relu(self.bn1(self.conv1(x))))
30        x = self.pool(F.relu(self.bn2(self.conv2(x))))
31        
32        x = x.view(-1, 64 * 8 * 8)
33        x = F.relu(self.fc1(x))
34        x = self.fc2(x)
35        return x
36
37# 消融实验对比：
38# 模型A（无BN）：基线模型
39# 模型B（有BN）：本模型
40# 
41# 预期结果（参见 {ref}`cnn-ablation-experiment`）：
42# - 有BN的模型收敛更快（约50%的训练时间）
43# - 有BN的模型允许使用更大学习率
44# - 准确率可能略有提升或持平
45# 
46# 实验方法：
47# 1. 训练两个模型各20个epoch
48# 2. 记录每轮的训练/测试准确率
49# 3. 绘制对比曲线，观察BN的影响

Dropout实现#

Dropout实现代码#
 1import torch.nn as nn
 2import torch.nn.functional as F
 3
 4class CNNWithDropout(nn.Module):
 5    """
 6    带Dropout正则化的CNN
 7    
 8    Dropout：训练时随机"关闭"部分神经元，防止过拟合
 9    参见 {doc}`../neural-network-basics/neural-training-basics` 中正则化讨论
10    """
11    def __init__(self, dropout_rate=0.5):
12        """
13        Args:
14            dropout_rate: 神经元被丢弃的概率（0.5表示50%）
15                         常见选择：0.3-0.5
16        """
17        super(CNNWithDropout, self).__init__()
18        self.conv1 = nn.Conv2d(3, 32, 3, padding=1)
19        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
20        self.pool = nn.MaxPool2d(2, 2)
21        self.fc1 = nn.Linear(64 * 8 * 8, 512)
22        self.fc2 = nn.Linear(512, 10)
23        
24        # Dropout层：只在全连接层使用是常见做法
25        # 卷积层通常不需要Dropout（参数量少，不易过拟合）
26        self.dropout = nn.Dropout(dropout_rate)
27        
28    def forward(self, x):
29        x = self.pool(F.relu(self.conv1(x)))
30        x = self.pool(F.relu(self.conv2(x)))
31        x = x.view(-1, 64 * 8 * 8)
32        
33        # 只在训练时应用Dropout（model.train()模式）
34        # 测试时自动关闭（model.eval()模式）
35        x = self.dropout(F.relu(self.fc1(x)))
36        
37        # 输出层不加Dropout
38        x = self.fc2(x)
39        return x
40
41# 消融实验对比（参见 {ref}`cnn-ablation-experiment`）：
42# 
43# | Dropout率 | 训练准确率 | 测试准确率 | 过拟合差距 |
44# |-----------|------------|------------|------------|
45# | 0.0（无）  | 95%        | 78%        | 17%        |
46# | 0.5（有）  | 89%        | 78%        | 11%        |
47# 
48# 关键观察：
49# - 无Dropout：训练准确率高但测试低 = 过拟合
50# - 有Dropout：训练准确率降低但测试不变 = 更好的泛化
51# 
52# 建议实验：
53# 1. 训练 dropout_rate=0.0 的模型
54# 2. 训练 dropout_rate=0.5 的模型
55# 3. 对比两者的 train/test 准确率差距
56# 4. 体会"正则化"的实际意义
