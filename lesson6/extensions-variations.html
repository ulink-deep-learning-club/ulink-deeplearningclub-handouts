自注意力机制#

自注意力模块实现#
 1import torch
 2import torch.nn as nn
 3import torch.nn.functional as F
 4
 5
 6class SelfAttention(nn.Module):
 7    """自注意力模块"""
 8    def __init__(self, in_channels):
 9        super(SelfAttention, self).__init__()
10        
11        self.query_conv = nn.Conv2d(in_channels, in_channels // 8, 1)
12        self.key_conv = nn.Conv2d(in_channels, in_channels // 8, 1)
13        self.value_conv = nn.Conv2d(in_channels, in_channels, 1)
14        
15        self.gamma = nn.Parameter(torch.zeros(1))
16        self.softmax = nn.Softmax(dim=-1)
17        
18    def forward(self, x):
19        batch_size, C, width, height = x.size()
20        
21        # 生成query, key, value
22        proj_query = self.query_conv(x).view(batch_size, -1, width * height).permute(0, 2, 1)
23        proj_key = self.key_conv(x).view(batch_size, -1, width * height)
24        proj_value = self.value_conv(x).view(batch_size, -1, width * height)
25        
26        # 计算注意力图
27        energy = torch.bmm(proj_query, proj_key)
28        attention = self.softmax(energy)
29        
30        # 应用注意力
31        out = torch.bmm(proj_value, attention.permute(0, 2, 1))
32        out = out.view(batch_size, C, width, height)
33        
34        # 残差连接
35        out = self.gamma * out + x
36        
37        return out
38
39
40if __name__ == "__main__":
41    # 测试SelfAttention
42    x = torch.randn(2, 64, 32, 32)
43    sa = SelfAttention(64)
44    y = sa(x)
45    print(f"Input shape: {x.shape}")
46    print(f"Output shape: {y.shape}")
47    print(f"SelfAttention parameters: {sum(p.numel() for p in sa.parameters())}")

多头注意力#

多头注意力模块实现#
 1import torch
 2import torch.nn as nn
 3
 4
 5class MultiHeadAttention(nn.Module):
 6    """多头注意力模块"""
 7    def __init__(self, in_channels, num_heads=8):
 8        super(MultiHeadAttention, self).__init__()
 9        
10        self.num_heads = num_heads
11        self.head_dim = in_channels // num_heads
12        
13        assert self.head_dim * num_heads == in_channels, "in_channels必须能被num_heads整除"
14        
15        # 为每个头创建独立的线性变换
16        self.qkv = nn.Conv2d(in_channels, in_channels * 3, 1)
17        self.proj = nn.Conv2d(in_channels, in_channels, 1)
18        
19    def forward(self, x):
20        batch_size, C, H, W = x.size()
21        
22        # 生成Q, K, V
23        qkv = self.qkv(x).reshape(batch_size, 3, self.num_heads, self.head_dim, H * W)
24        q, k, v = qkv[:, 0], qkv[:, 1], qkv[:, 2]
25        
26        # 缩放点积注意力
27        attn = (q @ k.transpose(-2, -1)) * (self.head_dim ** -0.5)
28        attn = attn.softmax(dim=-1)
29        
30        # 应用注意力
31        out = (attn @ v).transpose(1, 2).reshape(batch_size, C, H, W)
32        out = self.proj(out)
33        
34        return out
35
36
37if __name__ == "__main__":
38    # 测试MultiHeadAttention
39    x = torch.randn(2, 64, 32, 32)
40    mha = MultiHeadAttention(64, num_heads=8)
41    y = mha(x)
42    print(f"Input shape: {x.shape}")
43    print(f"Output shape: {y.shape}")
44    print(f"MultiHeadAttention parameters: {sum(p.numel() for p in mha.parameters())}")
