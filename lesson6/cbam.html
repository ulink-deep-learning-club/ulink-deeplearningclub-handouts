Convolutional Block Attention Module (CBAM)#

架构概述#

CBAM由Sanghyun Woo等人于2018年提出，结合了通道注意力和空间注意力，形成串行的注意力机制。

通道注意力模块#

CBAM的通道注意力模块与SE-Net类似，但使用了最大池化和平均池化的组合：

\[M_c(F) = \sigma(\text{MLP}(\text{AvgPool}(F)) + \text{MLP}(\text{MaxPool}(F)))\]
CBAM通道注意力模块实现#
 1import torch
 2import torch.nn as nn
 3
 4
 5class ChannelAttention(nn.Module):
 6    """CBAM的通道注意力模块"""
 7    def __init__(self, in_channels, reduction=16):
 8        super(ChannelAttention, self).__init__()
 9        
10        self.avg_pool = nn.AdaptiveAvgPool2d(1)
11        self.max_pool = nn.AdaptiveMaxPool2d(1)
12        
13        # 共享的MLP
14        self.mlp = nn.Sequential(
15            nn.Conv2d(in_channels, in_channels // reduction, 1, bias=False),
16            nn.ReLU(inplace=True),
17            nn.Conv2d(in_channels // reduction, in_channels, 1, bias=False)
18        )
19        
20        self.sigmoid = nn.Sigmoid()
21        
22    def forward(self, x):
23        # 平均池化和最大池化
24        avg_out = self.mlp(self.avg_pool(x))
25        max_out = self.mlp(self.max_pool(x))
26        
27        # 结合两种池化结果
28        out = avg_out + max_out
29        return self.sigmoid(out)
30
31
32if __name__ == "__main__":
33    # 测试ChannelAttention
34    x = torch.randn(2, 64, 32, 32)
35    ca = ChannelAttention(64, reduction=16)
36    y = ca(x)
37    print(f"Input shape: {x.shape}")
38    print(f"Attention shape: {y.shape}")
39    print(f"ChannelAttention parameters: {sum(p.numel() for p in ca.parameters())}")

空间注意力模块#

空间注意力模块通过通道维度的池化和卷积生成空间注意力图：

\[M_s(F) = \sigma(f^{7×7}([\text{AvgPool}(F); \text{MaxPool}(F)]))\]

其中 \(f^{7×7}\) 表示7×7卷积，\([·;·]\) 表示通道拼接。

CBAM空间注意力模块实现#
 1import torch
 2import torch.nn as nn
 3
 4
 5class SpatialAttention(nn.Module):
 6    """CBAM的空间注意力模块"""
 7    def __init__(self, kernel_size=7):
 8        super(SpatialAttention, self).__init__()
 9        
10        assert kernel_size in (3, 7), 'kernel size must be 3 or 7'
11        padding = 3 if kernel_size == 7 else 1
12        
13        self.conv = nn.Conv2d(2, 1, kernel_size, padding=padding, bias=False)
14        self.sigmoid = nn.Sigmoid()
15        
16    def forward(self, x):
17        # 沿通道维度进行平均池化和最大池化
18        avg_out = torch.mean(x, dim=1, keepdim=True)
19        max_out, _ = torch.max(x, dim=1, keepdim=True)
20        
21        # 拼接并卷积
22        x = torch.cat([avg_out, max_out], dim=1)
23        x = self.conv(x)
24        
25        return self.sigmoid(x)
26
27
28if __name__ == "__main__":
29    # 测试SpatialAttention
30    x = torch.randn(2, 64, 32, 32)
31    sa = SpatialAttention(kernel_size=7)
32    y = sa(x)
33    print(f"Input shape: {x.shape}")
34    print(f"Attention shape: {y.shape}")
35    print(f"SpatialAttention parameters: {sum(p.numel() for p in sa.parameters())}")

完整CBAM模块#

完整CBAM模块实现#
 1import torch
 2import torch.nn as nn
 3from .channel_attention import ChannelAttention
 4from .spatial_attention import SpatialAttention
 5
 6
 7class CBAM(nn.Module):
 8    """完整的CBAM模块"""
 9    def __init__(self, in_channels, reduction=16, kernel_size=7):
10        super(CBAM, self).__init__()
11        
12        self.channel_attention = ChannelAttention(in_channels, reduction)
13        self.spatial_attention = SpatialAttention(kernel_size)
14        
15    def forward(self, x):
16        # 先应用通道注意力
17        x = x * self.channel_attention(x)
18        
19        # 再应用空间注意力
20        x = x * self.spatial_attention(x)
21        
22        return x
23
24
25if __name__ == "__main__":
26    # 测试CBAM
27    x = torch.randn(2, 64, 32, 32)
28    cbam = CBAM(64, reduction=16, kernel_size=7)
29    y = cbam(x)
30    print(f"Input shape: {x.shape}")
31    print(f"Output shape: {y.shape}")
32    print(f"CBAM parameters: {sum(p.numel() for p in cbam.parameters())}")

CBAM-ResNet集成#

CBAM-ResNet基础块实现#
 1import torch
 2import torch.nn as nn
 3from .cbam import CBAM
 4
 5
 6class CBAMBasicBlock(nn.Module):
 7    """带有CBAM模块的ResNet基础块"""
 8    expansion = 1
 9    
10    def __init__(self, inplanes, planes, stride=1, downsample=None, reduction=16):
11        super(CBAMBasicBlock, self).__init__()
12        
13        self.conv1 = nn.Conv2d(inplanes, planes, kernel_size=3,
14                               stride=stride, padding=1, bias=False)
15        self.bn1 = nn.BatchNorm2d(planes)
16        
17        self.conv2 = nn.Conv2d(planes, planes, kernel_size=3,
18                               stride=1, padding=1, bias=False)
19        self.bn2 = nn.BatchNorm2d(planes)
20        
21        # CBAM模块
22        self.cbam = CBAM(planes, reduction)
23        
24        self.downsample = downsample
25        self.stride = stride
26        self.relu = nn.ReLU(inplace=True)
27        
28    def forward(self, x):
29        identity = x
30        
31        out = self.conv1(x)
32        out = self.bn1(out)
33        out = self.relu(out)
34        
35        out = self.conv2(out)
36        out = self.bn2(out)
37        
38        # 应用CBAM模块
39        out = self.cbam(out)
40        
41        if self.downsample is not None:
42            identity = self.downsample(x)
43            
44        out += identity
45        out = self.relu(out)
46        
47        return out
48
49
50if __name__ == "__main__":
51    # 测试CBAMBasicBlock
52    x = torch.randn(2, 64, 32, 32)
53    block = CBAMBasicBlock(64, 64)
54    y = block(x)
55    print(f"Input shape: {x.shape}")
56    print(f"Output shape: {y.shape}")
57    print(f"Block parameters: {sum(p.numel() for p in block.parameters())}")
